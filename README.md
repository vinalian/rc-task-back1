# Инструкция по запуску:
    Windows:
1. git clone *link*
2. python3 -m venv venv
venv/scripts/activate 
3. pip install -r req.txt 
4. python3 run.py 

    Linux:
1. git clone *link*
2. virtualenv env
3. virtualenv -p /usr/bin/python3 env
4. source virtualenv_name/bin/activate
